Most Useful Array Methods

Data:

const people = [
	{ name: "Sam", age: 30 },
	{ name: "Jane", age: 27 },
	{ name: "John", age: 32 },
	{ name: "Bill", age: 27 },
	{ name: "Andrew", age: 54 },
]

const arrays = [
	[1, 2], [3, 4], [[56], [78]]
]

const numbers = [1, 2, 3, 4]
const numbers2 = [5, 6, 7, 8]

const beasts = ['ant', 'bison', 'camel', 'duck', 'bison']
Docs: map() 

// create a new array of same length by going over all array elements, doing some action on them and returning result for each one

const new_array = numbers.map(element => element * 2)
// output: [ 2, 4, 6, 8 ]
Docs: filter() 

// create a new array with only elements that passed provided filter

const new_array = numbers.filter(element => element > 2)
// output: [ 3, 4 ]
Docs: reduce() 

// takes array of elements and reduce them to one value.
// First parameter is function that takes previously returned value and current value.
// Returned value is used in the next iteration

const result = people.reduce((groupedPeople, person) => {
	const age = person.age

	if (groupedPeople[age] == null) groupedPeople[age] = []

	groupedPeople[age].push(person)

	return groupedPeople
}, {})

/* output: {
	'27': [ { name: 'Jane', age: 27 }, { name: 'Bill', age: 27 } ],
	'30': [ { name: 'Sam', age: 30 } ],
	'32': [ { name: 'John', age: 32 } ],
	'54': [ { name: 'Andrew', age: 54 } ]
}*/
Docs: push() pop() unshift() shift() 

people.push(123)	// add element at the end of array
people.pop()		// remove element from the end of array
people.unshift(123)	// add element at the start of array
people.shift()		// remove element from the start of array
Docs: join() 

// returns a new string by concatenating all of the elements in this array, separated by a specified separator string

const result = numbers.join('-')
// output: '1-2-3-4'
Docs: split() 

// divides string into a list of substrings by searching for the pattern and puts these substrings into an array

const str = '1-2-3-4'
const new_array = str.split('-')
// output: [ '1', '2', '3', '4' ]
Docs: indexOf() lastIndexOf() 

// returns first index at which you can find provided value

const index = beasts.indexOf('bison')
// output: 1

// same as .indexOf(), but searches from the back

const index = beasts.lastIndexOf('bison')
// output: 4
Docs: find() findLast() 

// find first value in array that meets the requirement

const result = numbers.find(n => n < 2)
// output: 1

// find last value in the array that meets the requirement (same as .find(), but starts check from the back)

const result = numbers.findLast(n => n < 2)
// output: 1
Docs: slice() 

// return a (shallow) copy of an array between provided start and end indexes

const new_array = numbers.slice(2, 4)
// output: [ 3, 4 ]
Docs: splice() toSpliced() 

// removes elements from an array and, if necessary, inserts new elements in their place, returning the deleted elements
// parameters are (startIndex, deleteCount, items...)

numbers.splice(1, 0, 5)
// output: [ 1, 5, 2, 3, 4 ]

// same as .splice(), but doesn't modify original array

const new_array = numbers.toSpliced(1, 0, 5)
// output: [ 1, 5, 2, 3, 4 ]
Docs: sort() toSorted() 

// sort array elements based on condition

numbers.sort((a, b) => b - a)
// output: [ 4, 3, 2, 1 ]

// same as .sort(), but doesn't modify original array

const new_array = people.toSorted((a, b) => b - a)
// output: [ 4, 3, 2, 1 ]
Docs: includes() 

// return 'true' if the array has provided value, 'false' otherwise

const result = numbers.includes(4)
// output: true
Docs: reverse() toReversed() 

// reverse array elements

numbers.reverse()
// output: [ 4, 3, 2, 1 ]

// same as .reverse(), but doesn't modify original array

const new_array = numbers.toReversed()
// output: [ 4, 3, 2, 1 ]
Docs: flat() 

// takes nested arrays of arrays and flaten them into array with just those elements. Can provide how many levels to flaten

const new_array = arrays.flat(Number.POSITIVE_INFINITY)
// output: [ 1, 2, 3, 4, 56, 78 ]
Docs: concat() 

// return new array by combining two arrays

const new_array = numbers.concat(numbers2)
// output: [ 1, 2, 3, 4, 5, 6, 7, 8 ]
Docs: every() 

// returns 'true' if EVERY element in the array passed the condition, 'false' otherwise

const result = numbers.every(element => element < 40) 
// output: true
Docs: some() 

// return 'true' if AT LEAST ONE element in the array passed the condition, 'false' otherwise

const result = numbers.some(element => element < 40)
// output: true
Docs: Object.groupBy() 

// group array of objects based on a condition. (Can use Object or Map)
// this example groups objects based on first letter of the name

const new_object = Object.groupBy(people, person => person.name[0])
/* output: {
	S: [ {name: "Sam", age: 30} ],
	J: [ {name: "Jane", age: 27}, {name: "John", age: 32} ],
	B: [ {name: "Bill", age: 27} ],
	A: [ {name: "Andrew", age: 54} ],
}*/
Docs: with() 

// replaces value at specified index and return new array withou modifying original array

const new_array = people.with(1, 123)
// output: [ 1, 123, 3, 4 ]
Docs: at() 

// same as getting value from array with index [1], but allows to use negative numbers to get value from the end of array

const result = people.at(-1)
// output: 4
Docs: fill() 

// overwrite array elements between indexes 1 and 3 (not including) to value 123

numbers.fill(123, 1, 3)
// output: [ 1, 123, 123, 4 ]
